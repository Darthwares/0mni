```python
# agents/runtime/supervisor.py

from langgraph.graph import StateGraph
from anthropic import Anthropic

class SupervisorAgent:
    """Routes tasks to specialist agents"""

    def __init__(self, db_connection):
        self.db = db_connection
        self.anthropic = Anthropic()
        self.specialists = {
            "support": SupportAgent(db, "claude-opus-4.6"),
            "sales": SalesAgent(db, "gpt-5.2"),
            "recruiter": RecruiterAgent(db, "gemini-3.1-pro"),
            "code_review": CodeReviewAgent(db, "gemini-3.1-pro"),
            "docs": DocsAgent(db, "claude-opus-4.6"),
        }

    async def route_task(self, task):
        """Intelligent routing based on task type and context"""

        # Get full context
        context = await self.get_full_context(task)

        # Use Claude to route
        routing = await self.anthropic.messages.create(
            model="claude-opus-4.6",
            messages=[{
                "role": "user",
                "content": f"""
                Route this task to the best specialist:

                Task: {task.title}
                Type: {task.task_type}
                Context: {context}

                Available: {list(self.specialists.keys())}

                Return just the specialist name.
                """
            }],
            max_tokens=50
        )

        specialist_name = routing.content[0].text.strip()
        specialist = self.specialists[specialist_name]

        # Assign and execute
        await specialist.claim_and_execute(task)

    async def get_full_context(self, task):
        """Get ALL relevant context from unified database"""

        # Get entity (customer, candidate, deal, etc.)
        entity = self.db.get_entity(task.context_type, task.context_id)

        # Get ALL messages in this context
        messages = list(
            self.db.message()
            .context_type().filter(task.context_type)
            .context_id().filter(task.context_id)
            .iter()
        )

        # Get related tasks
        related_tasks = list(
            self.db.task()
            .context_id().filter(task.context_id)
            .iter()
        )

        # Get meetings/calls
        meetings = list(
            self.db.meeting()
            .iter()  # Filter in Python for simplicity
        )

        return {
            "entity": entity,
            "message_history": messages,
            "related_tasks": related_tasks,
            "meetings": [m for m in meetings if task.context_id in str(m)],
        }
```

**Specialist Agent with Self-Verification:**

```python
# agents/specialists/support_agent.py

class SupportAgent:
    """Handles customer support tasks with Claude Opus 4.6"""

    def __init__(self, db, model="claude-opus-4.6"):
        self.db = db
        self.anthropic = Anthropic()
        self.model = model
        self.agent_identity = None

    async def claim_and_execute(self, task):
        """Main agent loop with self-verification"""

        # 1. Claim the task
        self.db.reducers.claim_task(task.id)

        # 2. Get full context
        context = await self.get_context(task)

        # 3. Execute with LLM
        result = await self.execute_with_llm(task, context)

        # 4. Self-verify (2026 breakthrough!)
        verification = await self.self_verify(task, result, context)

        # 5. Complete with confidence score
        self.db.reducers.complete_task_with_verification(
            task_id=task.id,
            result=result["output"],
            confidence=verification["confidence"],
            thought_trace=result["reasoning_steps"]
        )

    async def execute_with_llm(self, task, context):
        """Execute task using Claude Opus 4.6"""

        customer = context["customer"]
        ticket = context["ticket"]
        history = context["messages"]

        prompt = f"""
        You are a customer support agent.

        Customer: {customer.name} ({customer.email})
        Sentiment: {customer.sentiment}

        Issue: {ticket.subject}

        Conversation History:
        {self.format_messages(history)}

        Task: {task.description}

        Provide:
        1. A helpful response
        2. Your reasoning steps
        3. Follow-up actions needed
        4. Your confidence (0.0-1.0)

        JSON response:
        {{
          "response": "...",
          "reasoning_steps": ["...", "..."],
          "follow_up_actions": [...],
          "confidence": 0.95
        }}
        """

        response = await self.anthropic.messages.create(
            model=self.model,
            messages=[{"role": "user", "content": prompt}],
            max_tokens=4096
        )

        return json.loads(response.content[0].text)

    async def self_verify(self, task, result, context):
        """Agent checks its own work"""

        verification_prompt = f"""
        Review your own response:

        Original issue: {context["ticket"].subject}
        Your response: {result["output"]}

        Self-check:
        1. Is it accurate and helpful?
        2. Any errors or omissions?
        3. Does it fully address the issue?
        4. Confidence level (0.0-1.0)?

        If confidence < 0.9, explain why.

        JSON:
        {{
          "passed": true/false,
          "confidence": 0.0-1.0,
          "issues_found": [...],
          "suggestion": "..."
        }}
        """

        verification = await self.anthropic.messages.create(
            model=self.model,
            messages=[{"role": "user", "content": verification_prompt}],
            max_tokens=1024
        )

        return json.loads(verification.content[0].text)
```

```rust
        status: TaskStatus::Unclaimed,

        ..Default::default()
    });

}

#[reducer]
pub fn claim_task(ctx: &ReducerContext, task_id: u64) -> Result<(), String> {
    let task = ctx.db.task().id().find(&task_id)
        .ok_or("Task not found")?;

    ctx.db.task().id().update(Task {
        assignee: Some(ctx.sender),
        status: TaskStatus::Claimed,
        claimed_at: Some(ctx.timestamp),
        ..task
    });

    Ok(())
}

#[reducer]
pub fn complete_task_with_verification(
    ctx: &ReducerContext,
    task_id: u64,
    result: String,
    confidence: f32,
    thought_trace: Vec<String>,
) -> Result<(), String> {
    let task = ctx.db.task().id().find(&task_id)
        .ok_or("Task not found")?;

    // Self-verification routing
    let status = if confidence >= 0.90 {
        TaskStatus::Completed  // High confidence → auto-complete
    } else if confidence >= 0.70 {
        TaskStatus::NeedsReview  // Medium → human review
    } else {
        TaskStatus::Escalated  // Low → immediate escalation
    };

    ctx.db.task().id().update(Task {
        status,
        ai_confidence: Some(confidence),
        self_verification_passed: Some(confidence >= 0.90),
        result: Some(result),
        thought_trace,
        completed_at: Some(ctx.timestamp),
        ..task
    });

    // Emit thought event for observability
    ctx.db.agent_thought_event().insert(AgentThoughtEvent {
        task_id,
        agent_id: ctx.sender,
        thought_type: ThoughtType::Completion,
        content: format!("Completed with {}% confidence", confidence * 100.0),
        confidence: Some(confidence),
        timestamp: ctx.timestamp,
    });

    Ok(())
}
```

## Risks & Mitigation

### Risk 1: AI API Costs Too High
**Mitigation**:
- Use smaller models where possible (Gemini for simple tasks)
- Implement aggressive caching
- Batch API calls
- Increase prices if needed (customers save $70k, can pay more)

### Risk 2: Self-Verification Not Reliable
**Mitigation**:
- Start with high thresholds (95% confidence)
- Gradual lowering as we validate
- Human review queue always available
- Track accuracy over time

### Risk 3: Customers Don't Trust AI Agents
**Mitigation**:
- Full transparency (thought traces)
- Human always in control
- Start with low-stakes tasks
- Show metrics proving reliability

### Risk 4: SpacetimeDB Doesn't Scale
**Mitigation**:
- SpacetimeDB team actively supporting
- Can shard by workspace if needed
- Self-hosted option for enterprise
- Worst case: migrate to Postgres (schema is simple)

### Risk 5: Incumbents Add AI Features
**Mitigation**:
- They're adding features, we're redesigning the platform
- Network effects from unified data
- Agent orchestration is hard (we have headstart)
- Move fast, lock in customers
